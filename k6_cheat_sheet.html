üåê HTTP Requests

GET request
let response = http.get('https://api.example.com/users');
POST request
let payload = JSON.stringify({ name: 'John' }); let params = { headers: { 'Content-Type': 'application/json' }, }; let response = http.post('https://api.example.com/users', payload, params);
PUT/PATCH/DELETE requests
http.put('https://api.example.com/users/1', payload, params); http.patch('https://api.example.com/users/1', payload, params); http.del('https://api.example.com/users/1');

‚úÖ Checks and Assertions

Basic response checks
import { check } from 'k6'; check(response, { 'status is 200': (r) => r.status === 200, 'response time < 500ms': (r) => r.timings.duration < 500, 'content type is JSON': (r) => r.headers['Content-Type'] === 'application/json', 'body contains user': (r) => r.body.includes('user'), });
JSON response validation
let data = JSON.parse(response.body); check(data, { 'user name is correct': (d) => d.name === 'John', 'user ID exists': (d) => d.id !== undefined, });

üìä Thresholds

Performance thresholds
export let options = { thresholds: { // 95% of requests must complete within 500ms http_req_duration: ['p(95)<500'], // Less than 1% of requests should fail http_req_failed: ['rate<0.01'], // Average response time should be under 200ms http_req_duration: ['avg<200'], // Custom counter threshold my_counter: ['count>100'], }, };

Advanced test scenarios

Using CSV data
import { SharedArray } from 'k6/data'; import papaparse from 'https://jslib.k6.io/papaparse/5.1.1/index.js'; const csvData = new SharedArray('users', function() { return papaparse.parse(open('./users.csv'), { header: true }).data; }); export default function() { let user = csvData[Math.floor(Math.random() * csvData.length)]; let response = http.get(`https://api.example.com/users/${user.id}`); }
Using JSON data
const data = JSON.parse(open('./data.json')); export default function() { let randomUser = data.users[Math.floor(Math.random() * data.users.length)]; }

Test lifecycle hooks

üö® Error Handling

Handling errors gracefully
import { check, fail } from 'k6'; export default function() { let response = http.get('https://api.example.com/users'); if (!check(response, { 'status is 200': (r) => r.status === 200, })) { fail('API returned non-200 status'); } try { let data = JSON.parse(response.body); } catch (e) { console.error('Failed to parse JSON:', e); } }

üîå WebSocket Testing

Testing WebSocket connections
import ws from 'k6/ws'; import { check } from 'k6'; export default function() { let url = 'ws://echo.websocket.org'; let params = { tags: { my_tag: 'websocket' } }; let response = ws.connect(url, params, function(socket) { socket.on('open', function() { console.log('WebSocket connected'); socket.send('Hello'); }); socket.on('message', function(data) { console.log('Received:', data); }); socket.setTimeout(function() { socket.close(); }, 10000); }); check(response, { 'status is 101': (r) => r && r.status === 101 }); }

üí° Pro Tips:
  • Start with smoke tests (1-2 VUs) before scaling up
  • Use sleep() to simulate realistic user behavior
  • Set appropriate thresholds to catch performance regressions
  • Use SharedArray for large datasets to save memory
  • Monitor system resources during load tests
  • Use scenarios for complex load patterns
  • Always validate your test results with checks
‚ö†Ô∏è Best Practices:
  • Never run load tests against production without permission
  • Start with low load and gradually increase
  • Monitor both application and infrastructure metrics
  • Use realistic test data and scenarios
  • Consider network latency in your test environment
